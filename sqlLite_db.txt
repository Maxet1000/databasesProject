CREATE TABLE "Game" (
	"gameID"	INTEGER NOT NULL,
	"title"	TEXT NOT NULL,
	"publisher"	TEXT NOT NULL,
	"consoleID"	INTEGER NOT NULL,
	"releaseDate"	DATE NOT NULL,
	"genreID"	INTEGER NOT NULL,
	"developerID"	INTEGER NOT NULL,
	FOREIGN KEY("consoleID") REFERENCES "Console"("consoleID"),
	FOREIGN KEY("developerID") REFERENCES "Developer"("developerID"),
	FOREIGN KEY("genreID") REFERENCES "Genre"("genreID"),
	PRIMARY KEY("gameID" AUTOINCREMENT)
);

CREATE TABLE "Console" (
	"consoleID"	INTEGER NOT NULL UNIQUE,
	"consoleName"	TEXT NOT NULL,
	"compatibleConsoleID"	INTEGER,
	FOREIGN KEY("compatibleConsoleID") REFERENCES "Console"("consoleID"),
	PRIMARY KEY("consoleID" AUTOINCREMENT)
);

CREATE TABLE "Copy" (
	"copyID"	INTEGER NOT NULL UNIQUE,
	"gameID"	INTEGER NOT NULL,
	"consoleID"	INTEGER NOT NULL,
	"customerID"	INTEGER,
	"purchasePrice"	INTEGER,
	"rentPrice"	INTEGER,
	"availability"	INTEGER,
	"dateOfReturn"	DATE,
	"warehouse"	TEXT NOT NULL,
	PRIMARY KEY("copyID" AUTOINCREMENT),
	FOREIGN KEY("customerID") REFERENCES "Customer"("customerID"),
	FOREIGN KEY("gameID") REFERENCES "Game"("gameID"),
	FOREIGN KEY("consoleID") REFERENCES "Console"("consoleID")
);

CREATE TABLE "Customer" (
	"customerID"	INTEGER NOT NULL UNIQUE,
	"customerName"	TEXT NOT NULL,
	"email"	TEXT NOT NULL,
	"balance"	INTEGER NOT NULL,
	PRIMARY KEY("customerID" AUTOINCREMENT)
);

CREATE TABLE "Developer" (
	"developerID"	INTEGER NOT NULL UNIQUE,
	"developerName"	TEXT NOT NULL,
	PRIMARY KEY("developerID" AUTOINCREMENT)
);

CREATE TABLE "Genre" (
	"genreID"	INTEGER NOT NULL UNIQUE,
	"genreName"	TEXT NOT NULL,
	PRIMARY KEY("genreID" AUTOINCREMENT)
);

CREATE TABLE "Transaction" (
	"transactionID"	INTEGER NOT NULL UNIQUE,
	"copyID"	INTEGER NOT NULL,
	"customerID"	INTEGER NOT NULL,
	"profit"	INTEGER NOT NULL,
	"time"	DATETIME NOT NULL,
	"transactionType"	TEXT NOT NULL,
	PRIMARY KEY("transactionID" AUTOINCREMENT),
	FOREIGN KEY("copyID") REFERENCES "Copy"("copyID"),
	FOREIGN KEY("customerID") REFERENCES "Customer"("customerID")
);
